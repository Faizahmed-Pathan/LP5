// Parallel BFS using OpenMP

#include <iostream>
#include <fstream>
#include <vector>
#include <queue>
#include <cstdlib>
#include <omp.h>
using namespace std;

void generateGraph(vector<vector<int>>& graph, int V, int edges) {
    srand(42);
    int count = 0;

    while (count < edges) {
        int u = rand() % V;
        int v = rand() % V;

        if (u != v) {
            graph[u].push_back(v);
            graph[v].push_back(u);
            count++;
        }
    }
}

void sequentialBFS(vector<vector<int>>& graph, int start) {
    int V = graph.size();
    vector<int> visited(V, 0);
    queue<int> q;

    visited[start] = 1;
    q.push(start);

    while (!q.empty()) {
        int node = q.front();
        q.pop();

        for (int neighbor : graph[node]) {
            if (!visited[neighbor]) {
                visited[neighbor] = 1;
                q.push(neighbor);
            }
        }
    }
}

void parallelBFS(vector<vector<int>>& graph, int start) {
    int V = graph.size();
    vector<int> visited(V, 0);
    vector<int> frontier, next_frontier;

    visited[start] = 1;
    frontier.push_back(start);

    while (!frontier.empty()) {
        next_frontier.clear();

        #pragma omp parallel for
        for (size_t i = 0; i < frontier.size(); i++) {
            int node = frontier[i];

            for (int neighbor : graph[node]) {
                bool add = false;

                #pragma omp critical
                {
                    if (!visited[neighbor]) {
                        visited[neighbor] = 1;
                        add = true;
                    }
                }

                if (add) {
                    #pragma omp critical
                    next_frontier.push_back(neighbor);
                }
            }
        }

        frontier = next_frontier;
    }
}

int main() {
    int sizes[] = {100, 500, 2000, 5000, 10000};
    int n = 5;

    int threads = omp_get_max_threads();

    cout << "Threads: " << threads << endl;
    cout << "Size\tSeqBFS\t\tParBFS\t\tSpeedup\t\tEfficiency" << endl;

    ofstream csv("bfs.csv");
    csv << "size,seq_bfs,par_bfs,speedup,efficiency\n";

    for (int i = 0; i < n; i++) {
        int V = sizes[i];
        int edges = V * 3;

        vector<vector<int>> graph(V);
        generateGraph(graph, V, edges);

        double start, seqTime, parTime;

        start = omp_get_wtime();
        sequentialBFS(graph, 0);
        seqTime = omp_get_wtime() - start;

        start = omp_get_wtime();
        parallelBFS(graph, 0);
        parTime = omp_get_wtime() - start;

        double speedup = seqTime / parTime;
        double efficiency = speedup / threads;

        cout << V << "\t" << seqTime << "\t" << parTime
             << "\t" << speedup << "\t" << efficiency << endl;

        csv << V << "," << seqTime << "," << parTime << ","
            << speedup << "," << efficiency << "\n";
    }

    csv.close();

    cout << "\nSaved to bfs.csv" << endl;

    return 0;
}
