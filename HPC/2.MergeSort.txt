// Compile: g++ -fopenmp 2_merge_sort.cpp -o 2_merge_sort
// Run: ./2_merge_sort

#include <iostream>
#include <fstream>
#include <cstdlib>
#include <cstring>
#include <omp.h>
using namespace std;

void merge(int *a, int l, int m, int r) {
    int n1 = m - l + 1;
    int n2 = r - m;

    int *L = new int[n1];
    int *R = new int[n2];

    for (int i = 0; i < n1; i++)
        L[i] = a[l + i];

    for (int i = 0; i < n2; i++)
        R[i] = a[m + 1 + i];

    int i = 0, j = 0, k = l;

    while (i < n1 && j < n2) {
        if (L[i] <= R[j])
            a[k++] = L[i++];
        else
            a[k++] = R[j++];
    }

    while (i < n1)
        a[k++] = L[i++];

    while (j < n2)
        a[k++] = R[j++];

    delete[] L;
    delete[] R;
}

void sequentialMergeSort(int *a, int l, int r) {
    if (l >= r)
        return;

    int m = (l + r) / 2;

    sequentialMergeSort(a, l, m);
    sequentialMergeSort(a, m + 1, r);

    merge(a, l, m, r);
}

void parallelMergeSort(int *a, int l, int r, int depth) {
    if (l >= r)
        return;

    int m = (l + r) / 2;

    if (depth > 0) {
        #pragma omp task
        parallelMergeSort(a, l, m, depth - 1);

        #pragma omp task
        parallelMergeSort(a, m + 1, r, depth - 1);

        #pragma omp taskwait
    } else {
        sequentialMergeSort(a, l, m);
        sequentialMergeSort(a, m + 1, r);
    }

    merge(a, l, m, r);
}

int main() {
    int sizes[] = {1000, 5000, 10000, 30000, 50000};
    int n = 5;

    int threads = omp_get_max_threads();

    cout << "Threads: " << threads << endl;
    cout << "Size\tSequential\tParallel\tSpeedup\t\tEfficiency" << endl;

    ofstream csv("merge_sort.csv");
    csv << "size,seq_merge,par_merge,speedup,efficiency\n";

    for (int t = 0; t < n; t++) {
        int sz = sizes[t];

        int *base = new int[sz];

        srand(42 + t);
        for (int i = 0; i < sz; i++) {
            base[i] = rand() % 100000;
        }

        int *a1 = new int[sz];
        int *a2 = new int[sz];

        memcpy(a1, base, sz * sizeof(int));
        memcpy(a2, base, sz * sizeof(int));

        double start;

        start = omp_get_wtime();
        sequentialMergeSort(a1, 0, sz - 1);
        double seqTime = omp_get_wtime() - start;

        start = omp_get_wtime();

        #pragma omp parallel
        {
            #pragma omp single
            {
                parallelMergeSort(a2, 0, sz - 1, 4);
            }
        }

        double parTime = omp_get_wtime() - start;

        double speedup = seqTime / parTime;
        double efficiency = speedup / threads;

        cout << sz << "\t" << seqTime << "\t" << parTime
             << "\t" << speedup << "\t" << efficiency << endl;

        csv << sz << "," << seqTime << "," << parTime << ","
            << speedup << "," << efficiency << "\n";

        delete[] base;
        delete[] a1;
        delete[] a2;
    }

    csv.close();

    cout << "\nSaved to merge_sort.csv" << endl;

    return 0;
}
